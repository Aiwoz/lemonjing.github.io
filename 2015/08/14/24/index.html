inline void swap(int &a, int &b)
{
    int tmp=a;
    a = b;
    b = tmp;
}

int partition(int p, int q)
{
    int r = rand() % (q - p + 1) + p;
    swap(pvalue[p], pvalue[r]);
    swap(v[p], v[r]);
    swap(w[p], w[r]);
    int i = p;
    int j = q + 1;
    while (1)
    {
        while (pvalue[++i] > pvalue[p]);
        while (pvalue[--j] < pvalue[p]);
        if (i >= j)
            break;
        swap(pvalue[i], pvalue[j]);
        swap(v[i], v[j]);
        swap(w[i], w[j]);
    }
    swap(pvalue[p], pvalue[j]);
    swap(v[p], v[j]);
    swap(w[p], w[j]);

    return j;
}

void random_qsort(int p, int q)
{
    if (p < q)
    {
        int r = partition(p, q);
        random_qsort(p, r - 1);
        random_qsort(r + 1, q);
    }
}
